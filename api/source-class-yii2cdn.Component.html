 71:     protected function buildSections ( array $config ) {

 73: 
 74:             if ( !array_key_exists($name, $config) || empty($config[$name]) ) {
 75:                 continue;
 76:             }
 77: 
 78:             $_attributes = isset ($config['fileAttrs'])
 79:                 ? isset($config['fileAttrs'][$name]) ? $config['fileAttrs'][$name] : []
 80:                 : [];
 81: 
 82:             // Create section(s) component
 83:             /** @var Section $section */            
 84:             $this->sections[$name] = \Yii::createObject($config['sectionClass'], [[
 85:                 'componentId' => $this->id,
 86:                 'section' =>$name,
 87:                 'files' => $config[$name],
 88:                 'baseUrl' => rtrim($this->baseUrl, '/'). "/{$name}",
 89:                 'attributes' => $_attributes,
 90:                 'fileClass' => $config['fileClass'],
 91:             ]]);
 92:         }
